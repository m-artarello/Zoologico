
drop table servicosrealizados;

drop table animal;
drop table profissional;

create table profissional (
	profissionalid SERIAL primary key,
	nome VARCHAR (100) not null,
	funcao VARCHAR (30) not null
);

create table animal (
	animalid SERIAL primary key,
	nome VARCHAR (50) not null,
	especie VARCHAR (20) not null,
	treinadorid SERIAL not null REFERENCES profissional (profissionalid)
);

create table servico (
	servicoid SERIAL primary key,
	descricao VARCHAR (75) not null
);

create table servicosrealizados (
	servicosrealizadosid  SERIAL primary key,
	servicoid SERIAL not null references servico (servicoid),
	animalid SERIAL not null references animal (animalid),
	datahora TIMESTAMP without time zone not null
);
